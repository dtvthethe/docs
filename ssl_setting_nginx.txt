1/ Create self-signed certificate:
$ sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout /xxx/nginx-ssl.key -out /xxx/nginx-ssl.crt

- Country Name (2 letter code) [AU]:VN
- State or Province Name (full name) [Some-State]:DANANG VN
- Locality Name (eg, city) []:DANANG
- Organization Name (eg, company) [Internet Widgits Pty Ltd]:BRANDIT . LTD                            
- Organizational Unit Name (eg, section) []:BRANDIT
- Common Name (e.g. server FQDN or YOUR name) []:brandit.test
- Email Address []:localhost@brandit.test 

2/ Configure NGINX to use SSL:
$ sudo nano /etc/nginx/snippets/self-signed.conf

ssl_certificate /home/dam.trieu.van.the/ssl_config/nginx-ssl.crt;
ssl_certificate_key /home/dam.trieu.van.the/ssl_config/nginx-ssl.key;

3/ Generate the dhparam.pem file:
$ sudo openssl dhparam -out /home/dam.trieu.van.the/ssl_config/dhparam.pem 2048

4/ Configuration snippet that points to our newly-generated SSL key and certificate.
$ sudo nano /etc/nginx/snippets/ssl-params.conf

ssl_protocols TLSv1.2;
ssl_prefer_server_ciphers on;
ssl_dhparam /home/dam.trieu.van.the/ssl_config/dhparam.pem;
ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-SHA384;
ssl_ecdh_curve secp384r1; # Requires nginx >= 1.1.0
ssl_session_timeout  10m;
ssl_session_cache shared:SSL:10m;
ssl_session_tickets off; # Requires nginx >= 1.5.9
# ssl_stapling on; # Requires nginx >= 1.3.7
# ssl_stapling_verify on; # Requires nginx => 1.3.7
resolver 8.8.8.8 8.8.4.4 valid=300s;
resolver_timeout 5s;
add_header X-Frame-Options DENY;
add_header X-Content-Type-Options nosniff;
add_header X-XSS-Protection "1; mode=block";

5/ Config site to use SSL:

$ sudo nano /etc/nginx/sites-available/www.brandit.test

server {
    listen 443 ssl;
    listen [::]:443 ssl;
    include snippets/self-signed.conf;
    include snippets/ssl-params.conf;

    server_name brandit.test www.brandit.test;

    root /var/www/html/brandit.test;
    index index.php;

}

server {
    listen 8000;
    listen [::]:8000;

    server_name brandit.test www.brandit.test;
    root /var/www/html/brandit.test;

    location / {
        try_files $uri /index.php$is_args$args;
    }

	location ~ ^/index\.php(/|$) {
        fastcgi_pass unix:/var/run/php/php-fpm.sock;
        fastcgi_split_path_info ^(.+\.php)(/.*)$;
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        fastcgi_param DOCUMENT_ROOT $realpath_root;

        internal;
    }

    location ~ \.php$ {
        return 404;
    }

    error_log /var/log/nginx/project_error.log;
    access_log /var/log/nginx/project_access.log;
}

$ sudo ln -s /etc/nginx/sites-available/www.brandit.test /etc/nginx/sites-enabled/
